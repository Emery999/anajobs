Web Scraping Strategy for Job Postings

Technical Implementation Guide for Non-Profit & Startup Career Pages

Executive Summary

This document outlines a comprehensive strategy for programmatically scraping job postings from 100+ websites in the non-profit, charity, and alternative energy sectors. The approach prioritizes efficiency through a hybrid methodology that combines fast programmatic crawling with browser automation fallbacks for complex sites.

Technical Architecture Overview

Recommended Hybrid Approach

Primary Method: Programmatic crawling (70-80% of sites)

  • Fast, lightweight HTTP requests
  • Static HTML parsing
  • Low resource consumption
  • High concurrency

Fallback Method: Browser automation (20-30% of sites)

  • Handles JavaScript-heavy sites
  • Manages dynamic content loading
  • Overcomes anti-bot measures

Core Technology Stack

Primary Stack:
- httpx/requests - HTTP client library
- BeautifulSoup4 - HTML parsing
- asyncio - Concurrent processing
- Python 3.8+ - Core language

Fallback Stack:
- Playwright - Browser automation
- Chrome/Chromium - Headless browser
- SQLite/JSON - Data persistence

def classify_site_complexity(response, soup):
    # Check for JavaScript frameworks
    js_frameworks = ['react', 'vue', 'angular', 'ember']
    if any(fw in response.text.lower() for fw in js_frameworks):
        return "browser_required"
    
    # Check for dynamic loading indicators
    dynamic_indicators = ['load-more', 'infinite-scroll', 'lazy-load']
    if soup.find(attrs={"class": dynamic_indicators}):
        return "browser_required"
    
    # Check for job content in static HTML
    job_content = soup.find_all(text=re.compile(r'\b(position|role|job|career)\b', re.I))
    if len(job_content) > 3:
        return "programmatic_sufficient"
    
    return "unknown"

Core Scraping Function

import httpx
import asyncio
from bs4 import BeautifulSoup
from playwright.async_api import async_playwright
import json
from datetime import datetime
import re

async def scrape_organization_jobs(url, session=None, browser=None):
    """
    Main scraping function with hybrid approach
    """
    result = {
        "organization": extract_org_name(url),
        "base_url": url,
        "jobs": [],
        "scrape_timestamp": datetime.now().isoformat(),
        "method_used": None,
        "status": "success",
        "error_message": None
    }
    
    try:
        # Attempt programmatic scraping first
        programmatic_result = await try_programmatic_scrape(url, session)
        if programmatic_result["success"]:
            result.update(programmatic_result)
            result["method_used"] = "programmatic"
            return result
        
        # Fallback to browser automation
        browser_result = await try_browser_scrape(url, browser)
        result.update(browser_result)
        result["method_used"] = "browser"
        return result
        
    except Exception as e:
        result["status"] = "error"
        result["error_message"] = str(e)
        return result

async def try_programmatic_scrape(url, session):
    """
    Fast programmatic scraping attempt
    """
    try:
        headers = {
            'User-Agent': 'Mozilla/5.0 (compatible; JobScraper/1.0)',
            'Accept': 'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8'
        }
        
        response = await session.get(url, headers=headers, follow_redirects=True)
        if response.status_code != 200:
            return {"success": False, "reason": f"HTTP {response.status_code}"}
        
        soup = BeautifulSoup(response.text, 'html.parser')
        
        # Find career page links
        career_links = find_career_links(soup, url)
        jobs = []
        
        # Scrape each career page
        for career_url in career_links[:5]:  # Limit to prevent infinite crawling
            career_response = await session.get(career_url, headers=headers)
            if career_response.status_code == 200:
                career_soup = BeautifulSoup(career_response.text, 'html.parser')
                page_jobs = extract_job_postings(career_soup, career_url)
                jobs.extend(page_jobs)
        
        return {"success": True, "jobs": jobs}
        
    except Exception as e:
        return {"success": False, "reason": str(e)}

async def try_browser_scrape(url, browser):
    """
    Browser automation fallback for complex sites
    """
    page = await browser.new_page()
    jobs = []
    
    try:
        await page.goto(url, wait_until='networkidle')
        
        # Find and click career links
        career_selectors = [
            'a[href*="career"]', 'a[href*="job"]', 
            'a[href*="employment"]', 'a[href*="work"]'
        ]
        
        for selector in career_selectors:
            links = await page.query_selector_all(selector)
            for link in links[:3]:  # Limit exploration
                try:
                    href = await link.get_attribute('href')
                    if href:
                        await page.goto(href, wait_until='networkidle')
                        content = await page.content()
                        soup = BeautifulSoup(content, 'html.parser')
                        page_jobs = extract_job_postings(soup, href)
                        jobs.extend(page_jobs)
                except:
                    continue  # Skip problematic links
        
        return {"success": True, "jobs": jobs}
        
    except Exception as e:
        return {"success": False, "reason": str(e)}
    finally:
        await page.close()

Job Posting Detection & Extraction

def extract_job_postings(soup, page_url):
    """
    Extract job postings from parsed HTML
    """
    jobs = []
    
    # Common job posting patterns
    job_containers = soup.find_all([
        lambda tag: tag.name in ['div', 'section', 'article'] and 
        any(keyword in tag.get_text().lower() for keyword in 
            ['position', 'role', 'job', 'opening', 'opportunity'])
    ])
    
    for container in job_containers:
        job = extract_single_job(container, page_url)
        if job and job['title']:  # Only add if we found a title
            jobs.append(job)
    
    return jobs

def extract_single_job(container, source_url):
    """
    Extract individual job posting details
    """
    job = {
        "title": "",
        "description": "",
        "requirements": "",
        "location": "",
        "employment_type": "",
        "application_url": "",
        "source_url": source_url,
        "posted_date": ""
    }
    
    # Title extraction
    title_tags = container.find_all(['h1', 'h2', 'h3', 'h4'])
    for tag in title_tags:
        text = tag.get_text().strip()
        if any(keyword in text.lower() for keyword in 
               ['position', 'role', 'job', 'manager', 'coordinator', 'specialist']):
            job['title'] = text
            break
    
    # Description extraction
    paragraphs = container.find_all('p')
    description_text = ' '.join([p.get_text().strip() for p in paragraphs[:3]])
    job['description'] = description_text[:1000]  # Limit length
    
    # Application link
    apply_link = container.find('a', string=re.compile(r'apply|join', re.I))
    if apply_link and apply_link.get('href'):
        job['application_url'] = apply_link['href']
    
    return job

def find_career_links(soup, base_url):
    """
    Find links to career/job pages
    """
    career_links = set([base_url])  # Always include main page
    
    # Find navigation links
    nav_links = soup.find_all('a', href=True)
    for link in nav_links:
        href = link['href'].lower()
        link_text = link.get_text().lower()
        
        if any(keyword in href or keyword in link_text for keyword in 
               ['career', 'job', 'employment', 'work', 'team', 'join']):
            full_url = urljoin(base_url, link['href'])
            career_links.add(full_url)
    
    return list(career_links)

async def scrape_all_organizations(url_list, output_file="job_scraping_results.json"):
    """
    Main function to scrape all organizations
    """
    results = []
    
    # Setup concurrent session and browser
    async with httpx.AsyncClient(timeout=30.0) as session:
        async with async_playwright() as p:
            browser = await p.chromium.launch(headless=True)
            
            # Process organizations concurrently (batch of 10)
            semaphore = asyncio.Semaphore(10)
            
            async def scrape_with_semaphore(url):
                async with semaphore:
                    return await scrape_organization_jobs(url, session, browser)
            
            # Execute scraping
            tasks = [scrape_with_semaphore(url) for url in url_list]
            results = await asyncio.gather(*tasks, return_exceptions=True)
            
            await browser.close()
    
    # Save results
    with open(output_file, 'w') as f:
        json.dump(results, f, indent=2, default=str)
    
    return results

Performance Expectations

Timing Estimates

  • Programmatic scraping: 2-5 seconds per site
  • Browser automation: 10-30 seconds per site
  • Total time for 100 sites: 15-45 minutes (depending on complexity mix)

Success Rate Projections

  • Programmatic success: 70-80% of sites
  • Browser automation success: 85-95% of remaining sites
  • Overall success rate: 90-95%

Resource Requirements

  • Memory: 2-4 GB RAM
  • CPU: Multi-core recommended for concurrency
  • Storage: 50-100 MB for job data output
  • Network: Stable broadband connection

Error Handling & Edge Cases

Common Challenges & Solutions

  1. Rate Limiting
    • Implement delays between requests
    • Rotate user agents
    • Use residential proxies if needed
  2. Dynamic Content
    • Browser automation fallback
    • Wait for specific elements to load
    • Screenshot failed attempts for debugging
  3. No Job Postings
    • Return empty jobs array, not failure
    • Log for manual review
    • Check volunteer/internship sections
  4. Blocked Requests
    • Retry with different headers
    • Use browser automation
    • Respect robots.txt
